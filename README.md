def sorting(numlist):
    
    print('given list',numlist, sep='\n')
    ascen=sorted(numlist)
    print('ascending list', ascen, sep='\n')
    descen=sorted(numlist, reverse=True)
    print('the descending order is', descen, sep='\n')
    
    
sorting([1,5,2,9,10,16,3,6,19,25,34])
